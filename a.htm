  10. public class BootCompleteReceiver extends BroadcastReceiver

  25.                         String protectedSim = sp.getString("simSerial", "");

  31.                                 //发送短信，有5个参数，第一个是要发送到的地址，第二个是发送人，可以设置为null，第三个是要发送的信息，第四个是发送状态，第五个是发送后的，都可以置为null
  32.                                 smsManager.sendTextMessage(number, null, "Sim卡已经变更了，手机可能被盗", null, null);
