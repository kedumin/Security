好啦，废话不多说，我们进入主题，今天我们要继续上一次讲的，就是手机被盗之后，获得手机所在的位置。

上一次，我们说到了，如果我们的sim卡被人换掉了，那么就会发送一条短信到我们的原来设定的安全号码那里的，而今天，我们就要获取到手机的位置啦，以便追踪回手机。

我们是这样设计的，我们通过发送一条短信，短信的内容就是一些要执行的操作的指令（比如说#*location*#），我们就在软件里面加入一个短信的广播，拦截到短信，然后分析是什么指令，然后就执行对应的操作，就像#*location*#这条指令一样，我们就可以理解为获取手机的位置。


所以我们要先新建一个类，用来提供手机的位置信息的

com.xiaobin.security.engine.GPSInfoProvider

  1. package com.xiaobin.security.engine;

  3. import android.content.Context;
  4. import android.content.SharedPreferences;
  5. import android.content.SharedPreferences.Editor;

  14. * @author Administrator

  16. */
  17. public class GPSInfoProvider
  18. {

  27.         }

  31.          * @return
  32.          */
  33.         public static synchronized GPSInfoProvider getInstance(Context context)
  34.         {
  35.                 if(gpsInfoProvider == null)
  36.                 {
  37.                         gpsInfoProvider = new GPSInfoProvider();
  38.                         GPSInfoProvider.context = context;
  39.                 }
  40.                 return gpsInfoProvider;
  41.         }
  42.        
  43.         public String getLocation()
  44.         {
  45.                 locationManager = (LocationManager) context.getSystemService(Context.LOCATION_SERVICE);
  46.                
  47.                 String provider = getBestProvider();
  48.                
  49.                 //这个方法是位置更新的的操作，有四个参数
  50.                 //第一个参数就是使用的定位设备啊，如gps，基站定位啊
  51.                 //第二个参数就是多长时间更新一次定位信息，太频繁了会很耗电，根据自己程序的实际需要来进行确定
  52.                 //第三个参数就是用户位移了多少米之后，就重新获取一次定位信息，太频繁了会很耗电，根据自己程序的实际需要来进行确定
  53.                 //最后一个参数就是在位置发生变化的回调方法
  54.                 locationManager.requestLocationUpdates(provider, 60000, 50, getListener());
  55.                 //locationManager.getAllProviders();//它会把所以支持的定位方式都打列出来，这样就可以知道手机所支持的定位啦
  56.                
  57.                 SharedPreferences sp = context.getSharedPreferences("config", Context.MODE_PRIVATE);
  58.                 String location = sp.getString("lostLocation", "");
  59.                 return location;
  60.         }
  61.        
  62.         //停止gps
  63.         public void stopGPSListener()
  64.         {
  65.                 if(locationManager != null)
  66.                 {
  67.                         locationManager.removeUpdates(getListener());
  68.                 }
  69.         }
  70.        
  71.         private String getBestProvider()
  72.         {
  73.                 Criteria criteria = new Criteria();
  74.                 //这个是定义它的定位精度的
  75.                 //Criteria.ACCURACY_COARSE  这个是一般的定位
  76.                 //Criteria.ACCURACY_FINE  这个是精准定位
  77.                 criteria.setAccuracy(Criteria.ACCURACY_FINE);
  78.                
  79.                 //设置是不是对海拔敏感的
  80.                 criteria.setAltitudeRequired(false);
  81.                
  82.                 //设置对手机的耗电量，定位要求越高，越耗电
  83.                 criteria.setPowerRequirement(Criteria.POWER_MEDIUM);
  84.                
  85.                 //设置对速度变化是不是敏感
  86.                 criteria.setSpeedRequired(true);
  87.                
  88.                 //设置在定位时，是不是允许与运营商交换数据的开销
  89.                 criteria.setCostAllowed(true);
  90.                
  91.                 //这个方法是用来得到最好的定位方式的，它有两个参数，一个是Criteria(类似于Map集合)，就是一些条件，比如说对加速度敏感啊，什么海拔敏感这些的
  92.                 //第二个参数就是，如果我们置为false，那么我们得到的也有可能是已经关掉了的设备，如果是true那么，就只会得到已经打开了的设备
  93.                 return locationManager.getBestProvider(criteria, true);
  94.         }
  95.        
  96.         //做成单例模式
  97.         private synchronized MyLocationListener getListener()
  98.         {
  99.                 if(listener == null)
  100.                 {
  101.                         listener = new MyLocationListener();
  102.                 }
  103.                 return listener;
  104.         }
  105.        
  106.         //=========================================================================
  107.        
  108.         private class MyLocationListener implements LocationListener
  109.         {

  111.                 @Override
  112.                 public void onLocationChanged(Location location)
  113.                 {
  114.                         //手机位置发生改变时调用的方法
  115.                         String latitude = "纬度：" + location.getLatitude();//纬度
  116.                         String longitude = "经度：" + location.getLongitude();//经度
  117.                         SharedPreferences sp = context.getSharedPreferences("config", Context.MODE_PRIVATE);
  118.                         Editor editor = sp.edit();
  119.                         editor.putString("lastLocation", latitude + " - " + longitude);
  120.                         editor.commit();
  121.                 }

  123.                 @Override
  124.                 public void onStatusChanged(String provider, int status, Bundle extras)
  125.                 {
  126.                         //定位设备的状态发生改变的时候调用的方法，比如说用户把设备打开，或关闭，第二个参数就是设备的状态啦
  127.                 }

  130.                 public void onProviderEnabled(String provider)
  131.                 {
  132.                         //设备打开的时候调用的方法
  133.                 }

  139.                 }
  140.                
  141.         }

  143. }

然后呢，我们就要开启一个广播接收者啦，一但拦截到我们发的短信，那么就进行指令的分析还有相应的处理
所以我们新建一个广播接收者com.xiaobin.security.receiver.SmsReceiver
  1. package com.xiaobin.security.receiver;

  3. import com.xiaobin.security.engine.GPSInfoProvider;

  5. import android.content.BroadcastReceiver;
  6. import android.content.Context;
  7. import android.content.Intent;
  8. import android.telephony.SmsManager;
  9. import android.telephony.SmsMessage;

  11. public class SmsReceiver extends BroadcastReceiver
  12. {

  14.         @Override
  15.         public void onReceive(Context context, Intent intent)
  16.         {
  17.                 Object[] pdus = (Object[]) intent.getExtras().get("pdus");
  18.                 for(Object pdu : pdus)
  19.                 {
  20.                         SmsMessage smsMessage = SmsMessage.createFromPdu((byte[]) pdu);
  21.                         //拿到短信内容
  22.                         String content = smsMessage.getMessageBody();
  23.                         //拿到发送人的电话号码
  24.                         String sender = smsMessage.getOriginatingAddress();
  25.                        
  26.                         //这个是通过短信发送指令，然后进行一些操作的
  27.                         if(content.equals("#*location*#"))
  28.                         {
  29.                                 abortBroadcast();//终止广播，免得让小偷看到短信
  30.                                 GPSInfoProvider gpsInfoProvider = GPSInfoProvider.getInstance(context);
  31.                                 String location = gpsInfoProvider.getLocation();
  32.                                 System.out.println(location);
  33.                                 if(!location.equals(""))
  34.                                 {
  35.                                         //发送短信
  36.                                         SmsManager smsManager = SmsManager.getDefault();
  37.                                         smsManager.sendTextMessage(sender, null, location, null, null);
  38.                                 }
  39.                         }
  40.                 }
  41.         }

  43. }

上面获取短信的代码可能大家有点疑问了，为什么会是pdus，这个的话，我们就可以下载一下Android自己写好了的短信管理器的代码啦，我们可以下载一下它的源码来看一下，Android源码下载的方法我们之前说过的啦！去看一下，就可以知道为什么的啦。大家可以搜一下sms就可以找到的啦，然后下来来看看吧。好啦，从上面的代码也可以看到，我们拿到了那条指令短信，然后，就获取位置，然后再发回给我们的手机，而且也把广播终止啦，这样，小偷就不会知道我们把短信发送到手机上啦，这样我们就可以瞒过小偷，进行一些操作啦，我们还可以定义多一些指令，比如说，删除东西的指令啦，格式化的指令啦，乱发东西的指令啦，都可以的，我们看一下有没有空，尽量写多一些功能!好啦，现在广播接收者就写好啦，我们也要在AndroidMainfest里面注册它啦
  1. <receiver

  3.             <intent-filter android:priority="1000">
  4.                 <action android:name="android.provider.Telephony.SMS_RECEIVED" />
  5.             </intent-filter>
  6.         </receiver>
